ANGULARJS Конспект
====================
1) ДИРЕКТИВЫ
    //основные
    - ng-app = "" // ссылка на модуль ангуляра, который используется
    - ng-controller = ""// ссылка на контроллер, переменные и методы которого будут доступны внутри тега с таким атрибутом
    //одностороннее связывание
    - ng-bind = "" // ссылка на переменную из scope, значение которой вставится внутрьь тега
    - ng-bind-template = "" // то же самое, но можно вставить не переменную, а целое выражение
    - {{ }} // то же самое, но гораздо круче, т.к. короче и можно вставлять в любое место (не только внутрь тега)
    - ng-non-bindable // отключить биндинг (т.е. если в элементе присутствуют {{ }}, они будут игнорироваться)
    //двустороннее связывание
    - ng-model = "имя_переменной_из_scope" -- связать данные в тэге с переменной и переменную с данными в тэге 
        (доступно только для input, textarea, select, т.е. для элементов, допускающих пользовательский ввод)
    -

    //остальное
    - ng-repeat = "item in items as (key, value)" // повторить тэг с этим атрибутом столько раз, сколько элементов в переданном массиве/объекте items  
    - ng-repeat-start = "item in items", ng-repeat-end // аналогично, но позволяет повторять несколько тэгов без обертки
        //доступные переменные в директиве repeat: 
        a) $index -- текущий номер итерации
        б) $even -- четная итерация (булевый)
        в) $odd -- нечетная итерации (булевый)
        //виды перебора:
        а) item in items // итерация по элементам массива
        б) property in item // итерация по свойствам объекта
        в) (key, value) in item //итерация по свойствам объекта (с ключами)
    - ng-cloak //не показывать элемент (== display: none) до тех пор, пока не загрузятся данные из переменных angular (напр. чтобы не было видно фигурных скобок)
    - ng-if="условие" //элемент будет показан, если условие == true и скрыт, если условие == false
    - ng-class="$scope.isActive ? 'active' : ''" // присваивает класс в зависимости от условия
    - ng-options="item for item in items" // аналог ng-repeat, но только генерирует элементы option выпадающего списка
        //подробнее значение атрибута выглядит так: item.prop1 for item in items track by item.prop2
                                        что отображается в селекте  что берется за основу  что идет в value

    //частичное представление -- фрагмент html-кода, который мы инклудим на страницу с помощью аякса. Пример:
    1) <ng-include src="table.html">сюда вставится содержимое файла table.html</ng-include>
    2) <div ng-include="table.html">аналогично</div>
    3) <div class="ng-include: 'table.html';">аналогично</div>

    //свитч (переключение шаблонов в зависимости от условий без подгрузки аяксом). Пример:
    <div ng-switch on="mode">
        <div ng-switch-on="table"></div> //покажется, если mode == table
        <div ng-switch-on="list"></div>
        <div ng-switch-on="graph"></div>
    </div>
